-- 配置开关（按钮会覆盖这些初始值，默认为关闭）
getgenv().RemoveAccessories = false   -- 饰品剔除开关
getgenv().FakeHeadlessKorblox = false -- 虚假头/腿效果开关

-- 等待游戏加载
repeat task.wait() until game:IsLoaded()

local Players = game:GetService("Players")
local player = Players.LocalPlayer
local StarterGui = game:GetService("StarterGui")
local UserInputService = game:GetService("UserInputService")

-- 资源ID（来自原脚本）
local HEAD_MESH_ID = "134082579"
local HEAD_TEX_ID = "134082627"
local LEG_MESH_ID = "101851696"
local LEG_TEX_ID = "101851254"

-- 用于存储当前角色备份数据的容器名称（隐藏）
local BACKUP_FOLDER_NAME = "__EffectBackup"

-- ========== 工具函数（增强版：备份/恢复部件属性）==========
local function getBackupFolder(character)
    local folder = character:FindFirstChild(BACKUP_FOLDER_NAME)
    if not folder then
        folder = Instance.new("Folder")
        folder.Name = BACKUP_FOLDER_NAME
        folder.Parent = character
    end
    return folder
end

-- 备份部件原始属性（颜色、材质、透明度）
local function backupProperties(part, backupFolder, propsName)
    -- 移除旧的同名备份（避免残留）
    local old = backupFolder:FindFirstChild(propsName)
    if old then old:Destroy() end

    local propsFolder = Instance.new("Folder")
    propsFolder.Name = propsName
    propsFolder.Parent = backupFolder

    local colorVal = Instance.new("StringValue")
    colorVal.Name = "BrickColor"
    colorVal.Value = part.BrickColor.Name
    colorVal.Parent = propsFolder

    local materialVal = Instance.new("StringValue")
    materialVal.Name = "Material"
    materialVal.Value = part.Material.Name
    materialVal.Parent = propsFolder

    local transVal = Instance.new("NumberValue")
    transVal.Name = "Transparency"
    transVal.Value = part.Transparency
    transVal.Parent = propsFolder
end

-- 恢复部件原始属性
local function restoreProperties(part, backupFolder, propsName)
    local propsFolder = backupFolder:FindFirstChild(propsName)
    if propsFolder then
        local colorVal = propsFolder:FindFirstChild("BrickColor")
        if colorVal then
            part.BrickColor = BrickColor.new(colorVal.Value)
        end
        local materialVal = propsFolder:FindFirstChild("Material")
        if materialVal then
            part.Material = Enum.Material[materialVal.Value]
        end
        local transVal = propsFolder:FindFirstChild("Transparency")
        if transVal then
            part.Transparency = transVal.Value
        end
        propsFolder:Destroy() -- 用完即删
    end
end

-- 移动实例到备份文件夹（用于贴花/网格）
local function moveToBackup(instance, backupFolder)
    instance.Parent = backupFolder
end

-- ========== 饰品剔除功能 ==========
local function hideAccessories(character)
    if not character then return end
    local function hide(inst)
        if inst:IsA("Accessory") then
            local handle = inst:FindFirstChild("Handle")
            if handle and handle:IsA("BasePart") then
                handle.Transparency = 1
                for _, d in ipairs(handle:GetDescendants()) do
                    if d:IsA("Decal") then d.Transparency = 1 end
                end
            end
        elseif inst:IsA("Hat") or inst:IsA("Tool") then
            local handle = inst:FindFirstChild("Handle")
            if handle and handle:IsA("BasePart") then
                handle.Transparency = 1
            end
        end
    end
    for _, obj in ipairs(character:GetDescendants()) do
        hide(obj)
    end
end

local function showAccessories(character)
    if not character then return end
    local function show(inst)
        if inst:IsA("Accessory") then
            local handle = inst:FindFirstChild("Handle")
            if handle and handle:IsA("BasePart") then
                handle.Transparency = 0
                for _, d in ipairs(handle:GetDescendants()) do
                    if d:IsA("Decal") then d.Transparency = 0 end
                end
            end
        elseif inst:IsA("Hat") or inst:IsA("Tool") then
            local handle = inst:FindFirstChild("Handle")
            if handle and handle:IsA("BasePart") then
                handle.Transparency = 0
            end
        end
    end
    for _, obj in ipairs(character:GetDescendants()) do
        show(obj)
    end
end

-- ========== 虚假头/腿效果功能（带完整属性备份与恢复）==========
local function applyHeadEffect(character)
    if not character then return end
    local head = character:FindFirstChild("Head")
    if not head then return end

    local backup = getBackupFolder(character)

    -- 备份头部原始属性
    backupProperties(head, backup, "HeadProps")

    -- 备份并移除头部原有的 Decal 和 SpecialMesh
    for _, child in ipairs(head:GetChildren()) do
        if child:IsA("Decal") or child:IsA("SpecialMesh") then
            moveToBackup(child, backup)
        end
    end

    -- 添加虚假头部网格
    local mesh = Instance.new("SpecialMesh")
    mesh.MeshType = Enum.MeshType.FileMesh
    mesh.MeshId = "rbxassetid://" .. HEAD_MESH_ID
    mesh.TextureId = "rbxassetid://" .. HEAD_TEX_ID
    mesh.Scale = Vector3.new(1.25, 1.25, 1.25)
    mesh.Offset = Vector3.new(0, 0, 0)
    mesh.Parent = head
    mesh.Name = "__FakeHeadMesh"

    -- 修改头部视觉效果（虚假头效果）
    head.Transparency = 0.1
    head.BrickColor = BrickColor.new("Really black")
    head.Material = Enum.Material.Plastic
end

local function applyLegEffect(character)
    if not character then return end
    local leg = character:FindFirstChild("Right Leg") or character:FindFirstChild("RightLowerLeg")
    if not leg then return end

    local backup = getBackupFolder(character)

    -- 备份原腿原始属性
    backupProperties(leg, backup, "LegProps")

    -- 备份并移除腿部原有的 SpecialMesh
    for _, child in ipairs(leg:GetChildren()) do
        if child:IsA("SpecialMesh") then
            moveToBackup(child, backup)
        end
    end

    -- 创建假腿（原脚本 v36 方式）
    local fakeLeg = Instance.new("Part")
    fakeLeg.Name = "FakeKorbloxLeg"
    fakeLeg.Size = leg.Size
    fakeLeg.CFrame = leg.CFrame
    fakeLeg.Anchored = false
    fakeLeg.CanCollide = false
    fakeLeg.Transparency = 0
    fakeLeg.BrickColor = BrickColor.new("Really black")
    fakeLeg.Material = Enum.Material.Plastic

    local mesh = Instance.new("SpecialMesh")
    mesh.MeshType = Enum.MeshType.FileMesh
    mesh.MeshId = "rbxassetid://" .. LEG_MESH_ID
    mesh.TextureId = "rbxassetid://" .. LEG_TEX_ID
    mesh.Scale = Vector3.new(1, 1, 1)
    mesh.Offset = Vector3.new(0, 0, 0)
    mesh.Parent = fakeLeg

    fakeLeg.Parent = character
    fakeLeg.Name = "__FakeLegPart"

    local weld = Instance.new("WeldConstraint")
    weld.Part0 = leg
    weld.Part1 = fakeLeg
    weld.Parent = fakeLeg

    -- 隐藏原腿（但保留其属性备份，用于恢复）
    leg.Transparency = 1
    -- 注意：不修改原腿颜色/材质，仅透明
end

local function removeHeadEffect(character)
    if not character then return end
    local head = character:FindFirstChild("Head")
    if not head then return end

    local backup = character:FindFirstChild(BACKUP_FOLDER_NAME)
    if not backup then return end

    -- 恢复头部原始属性
    restoreProperties(head, backup, "HeadProps")

    -- 删除我们添加的虚假网格
    for _, child in ipairs(head:GetChildren()) do
        if child.Name == "__FakeHeadMesh" then
            child:Destroy()
        end
    end

    -- 从备份中恢复原有的 Decal 和 Mesh
    for _, child in ipairs(backup:GetChildren()) do
        -- 这些是之前被移动的 Decal/Mesh，直接设回头部
        if child:IsA("Decal") or child:IsA("SpecialMesh") then
            child.Parent = head
        end
    end
    -- 注意：上面循环会移动所有备份的子项，但可能包含属性文件夹，不过属性文件夹在 restoreProperties 时已被销毁，所以安全。
    -- 但如果有其他备份项（例如来自腿部的），需要过滤。简单处理：只移动 Decal 和 SpecialMesh
    -- 更好的做法是在备份时单独存放，但这里为了简洁，我们只移动特定类型。
    -- 我们已经用 restoreProperties 处理了属性，所以属性文件夹已被销毁。
end

local function removeLegEffect(character)
    if not character then return end
    local leg = character:FindFirstChild("Right Leg") or character:FindFirstChild("RightLowerLeg")
    if not leg then return end

    local backup = character:FindFirstChild(BACKUP_FOLDER_NAME)
    if not backup then return end

    -- 恢复原腿原始属性（包括透明度、颜色、材质）
    restoreProperties(leg, backup, "LegProps")

    -- 删除假腿及其焊接
    for _, child in ipairs(character:GetChildren()) do
        if child.Name == "__FakeLegPart" then
            child:Destroy()
        end
    end

    -- 从备份中恢复原有的 Mesh（如果有）
    for _, child in ipairs(backup:GetChildren()) do
        if child:IsA("SpecialMesh") then
            child.Parent = leg
        end
    end
end

local function applyAllEffects(character)
    applyHeadEffect(character)
    applyLegEffect(character)
end

local function removeAllEffects(character)
    removeHeadEffect(character)
    removeLegEffect(character)
end

-- ========== 通知函数 ==========
local function notify(title, text, duration)
    StarterGui:SetCore("SendNotification", {
        Title = title,
        Text = text,
        Duration = duration or 5
    })
end

-- ========== GUI 创建与按键控制 ==========
local function setupGUI()
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "EffectToggleGUI"
    screenGui.ResetOnSpawn = false
    screenGui.Parent = player:WaitForChild("PlayerGui")

    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 200, 0, 150)
    frame.Position = UDim2.new(0, 500, 0, 10)
    frame.BackgroundColor3 = Color3.new(0.2, 0.2, 0.2)
    frame.BackgroundTransparency = 0.3
    frame.BorderSizePixel = 2
    frame.BorderColor3 = Color3.new(1, 1, 1)
    frame.Active = true
    frame.Draggable = true
    frame.Parent = screenGui

    local title = Instance.new("TextLabel")
    title.Size = UDim2.new(1, 0, 0, 25)
    title.Position = UDim2.new(0, 0, 0, 0)
    title.BackgroundTransparency = 1
    title.Text = "效果切换"
    title.TextColor3 = Color3.new(1, 1, 1)
    title.TextScaled = true
    title.Font = Enum.Font.SourceSansBold
    title.Parent = frame

    -- 按钮1：饰品剔除
    local btn1 = Instance.new("TextButton")
    btn1.Size = UDim2.new(0.9, 0, 0, 30)
    btn1.Position = UDim2.new(0.05, 0, 0, 30)
    btn1.BackgroundColor3 = Color3.new(0.3, 0.6, 1)
    btn1.Text = "饰品剔除 (关)"
    btn1.TextColor3 = Color3.new(1, 1, 1)
    btn1.Parent = frame

    -- 按钮2：无头断腿
    local btn2 = Instance.new("TextButton")
    btn2.Size = UDim2.new(0.9, 0, 0, 30)
    btn2.Position = UDim2.new(0.05, 0, 0, 65)
    btn2.BackgroundColor3 = Color3.new(0.3, 0.6, 1)
    btn2.Text = "无头断腿 (关)"
    btn2.TextColor3 = Color3.new(1, 1, 1)
    btn2.Parent = frame

    -- 按钮3：同时切换
    local btn3 = Instance.new("TextButton")
    btn3.Size = UDim2.new(0.9, 0, 0, 30)
    btn3.Position = UDim2.new(0.05, 0, 0, 100)
    btn3.BackgroundColor3 = Color3.new(0.8, 0.3, 0.3)
    btn3.Text = "一起开关"
    btn3.TextColor3 = Color3.new(1, 1, 1)
    btn3.Parent = frame

    -- 状态变量
    local accessoryState = false
    local effectState = false

    -- 更新按钮文字
    local function updateButtons()
        btn1.Text = "饰品剔除 " .. (accessoryState and "(开)" or "(关)")
        btn2.Text = "无头断腿 " .. (effectState and "(开)" or "(关)")
    end

    -- 定义一起开关的函数（供 T 键调用）
    local function toggleBoth()
        local char = player.Character
        if not char then return end

        if accessoryState == effectState then
            -- 相同则一起翻转
            if accessoryState then
                showAccessories(char)
                removeAllEffects(char)
            else
                hideAccessories(char)
                applyAllEffects(char)
            end
            accessoryState = not accessoryState
            effectState = not effectState
        else
            -- 不同则强制都打开
            if not accessoryState then
                hideAccessories(char)
                accessoryState = true
            end
            if not effectState then
                applyAllEffects(char)
                effectState = true
            end
        end
        updateButtons()
    end

    -- 按钮1回调
    btn1.MouseButton1Click:Connect(function()
        local char = player.Character
        if not char then return end

        if accessoryState then
            showAccessories(char)
            accessoryState = false
        else
            hideAccessories(char)
            accessoryState = true
        end
        updateButtons()
    end)

    -- 按钮2回调
    btn2.MouseButton1Click:Connect(function()
        local char = player.Character
        if not char then return end

        if effectState then
            removeAllEffects(char)
            effectState = false
        else
            applyAllEffects(char)
            effectState = true
        end
        updateButtons()
    end)

    -- 按钮3回调
    btn3.MouseButton1Click:Connect(toggleBoth)

    -- 角色重生时同步状态
    player.CharacterAdded:Connect(function(character)
        task.wait()
        if accessoryState then
            hideAccessories(character)
        end
        if effectState then
            applyAllEffects(character)
        end
        updateButtons()
    end)

    -- 如果已有角色，立即应用初始状态（根据getgenv变量）
    if player.Character then
        task.defer(function()
            if getgenv().RemoveAccessories then
                hideAccessories(player.Character)
                accessoryState = true
            end
            if getgenv().FakeHeadlessKorblox then
                applyAllEffects(player.Character)
                effectState = true
            end
            updateButtons()
        end)
    end

    updateButtons()

    -- ===== 添加按键控制 =====
    UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end

        if input.KeyCode == Enum.KeyCode.T then
            -- T 键：显示/隐藏 GUI
            screenGui.Enabled = not screenGui.Enabled
        elseif input.KeyCode == Enum.KeyCode.R then
            -- R 键：一键开关（同时切换两个效果）
            toggleBoth()
        end
    end)

    return screenGui
end

-- 启动 GUI
setupGUI()
